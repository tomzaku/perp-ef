The Factory Pattern provides an interface for creating objects without specifying their exact class. It encapsulates object creation logic and lets the caller request an object by type or configuration.

**Factory Method:** A single function/method that returns different objects based on input.
**Abstract Factory:** A factory of factories — creates families of related objects.

**Frontend use cases:**
- Creating different UI components based on configuration
- API client factory (REST vs GraphQL vs mock)
- Notification factory (email, SMS, push)
- Form field factory (text, select, checkbox based on schema)

## Solution

```js
// ============================
// Factory Method Pattern
// ============================
class Button {
  constructor(text, variant) { this.text = text; this.variant = variant; }
  render() { return `<button class="btn-${this.variant}">${this.text}</button>`; }
}

class IconButton {
  constructor(icon, text, variant) { this.icon = icon; this.text = text; this.variant = variant; }
  render() { return `<button class="btn-${this.variant}"><i class="${this.icon}"></i> ${this.text}</button>`; }
}

class LinkButton {
  constructor(text, href) { this.text = text; this.href = href; }
  render() { return `<a href="${this.href}" class="btn-link">${this.text}</a>`; }
}

function createButton(config) {
  switch (config.type) {
    case 'icon':
      return new IconButton(config.icon, config.text, config.variant || 'primary');
    case 'link':
      return new LinkButton(config.text, config.href);
    default:
      return new Button(config.text, config.variant || 'primary');
  }
}

console.log(createButton({ type: 'default', text: 'Submit' }).render());
console.log(createButton({ type: 'icon', icon: 'fa-save', text: 'Save' }).render());
console.log(createButton({ type: 'link', text: 'Docs', href: '/docs' }).render());

// ============================
// Abstract Factory — themed UI kits
// ============================
function createLightTheme() {
  return {
    createButton(text) {
      return { render: () => `<button class="light-btn">${text}</button>` };
    },
    createInput(placeholder) {
      return { render: () => `<input class="light-input" placeholder="${placeholder}" />` };
    },
    createCard(content) {
      return { render: () => `<div class="light-card">${content}</div>` };
    }
  };
}

function createDarkTheme() {
  return {
    createButton(text) {
      return { render: () => `<button class="dark-btn">${text}</button>` };
    },
    createInput(placeholder) {
      return { render: () => `<input class="dark-input" placeholder="${placeholder}" />` };
    },
    createCard(content) {
      return { render: () => `<div class="dark-card">${content}</div>` };
    }
  };
}

function buildUI(themeFactory) {
  const btn = themeFactory.createButton('Click me');
  const input = themeFactory.createInput('Search...');
  const card = themeFactory.createCard('Hello World');
  return [btn, input, card].map(c => c.render()).join('\n');
}

console.log(buildUI(createLightTheme()));
console.log(buildUI(createDarkTheme()));

// ============================
// Practical: API Client Factory
// ============================
function createApiClient(type = 'rest') {
  if (type === 'rest') {
    return {
      async get(url) { return fetch(url).then(r => r.json()); },
      async post(url, data) {
        return fetch(url, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(data),
        }).then(r => r.json());
      }
    };
  }
  if (type === 'mock') {
    const store = {};
    return {
      async get(url) { return store[url] || null; },
      async post(url, data) { store[url] = data; return { success: true, data }; }
    };
  }
  throw new Error(`Unknown client type: ${type}`);
}

const api = createApiClient('mock');
api.post('/users', { name: 'Alice' }).then(console.log);
```
