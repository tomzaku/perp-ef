The Facade Pattern provides a simplified interface to a complex subsystem. It doesn't add new functionality — it makes existing functionality easier to use by hiding complexity behind a clean API.

**Frontend use cases:**
- Wrapping complex browser APIs (Web Audio, Canvas, WebGL)
- Unifying multiple microservice calls behind a single function
- Simplifying a complex library's API for common use cases
- jQuery is essentially a facade over the DOM API

**When to use:**
- When a subsystem has many interdependent classes
- When you want a simple default interface for common tasks
- When you want to decouple client code from subsystem details

## Solution

```js
// ============================
// Facade over multiple API services
// ============================
class UserApi {
  async getProfile(id) { return { id, name: 'Alice', email: 'alice@test.com' }; }
}

class OrderApi {
  async getOrders(userId) { return [{ id: 1, total: 99.99 }]; }
}

class NotificationApi {
  async getUnread(userId) { return [{ id: 1, message: 'Welcome!' }]; }
}

class PreferencesApi {
  async getPrefs(userId) { return { theme: 'dark', lang: 'en' }; }
}

// Facade: one call instead of four
class UserDashboardFacade {
  constructor() {
    this.userApi = new UserApi();
    this.orderApi = new OrderApi();
    this.notifApi = new NotificationApi();
    this.prefsApi = new PreferencesApi();
  }

  async getDashboardData(userId) {
    const [profile, orders, notifications, preferences] = await Promise.all([
      this.userApi.getProfile(userId),
      this.orderApi.getOrders(userId),
      this.notifApi.getUnread(userId),
      this.prefsApi.getPrefs(userId),
    ]);

    return {
      profile,
      orders,
      notifications,
      preferences,
      summary: {
        totalOrders: orders.length,
        unreadCount: notifications.length,
        displayName: profile.name,
      }
    };
  }
}

const dashboard = new UserDashboardFacade();
dashboard.getDashboardData(1).then(data => {
  console.log(data.summary);
  // { totalOrders: 1, unreadCount: 1, displayName: 'Alice' }
});

// ============================
// DOM Manipulation Facade (mini jQuery)
// ============================
const $ = {
  select(selector) {
    return document.querySelector(selector);
  },
  selectAll(selector) {
    return [...document.querySelectorAll(selector)];
  },
  on(selector, event, handler) {
    const el = typeof selector === 'string' ? this.select(selector) : selector;
    if (el) el.addEventListener(event, handler);
    return () => el?.removeEventListener(event, handler);
  },
  css(selector, styles) {
    const el = typeof selector === 'string' ? this.select(selector) : selector;
    if (el) Object.assign(el.style, styles);
  },
  addClass(selector, ...classes) {
    const el = typeof selector === 'string' ? this.select(selector) : selector;
    if (el) el.classList.add(...classes);
  },
  hide(selector) { this.css(selector, { display: 'none' }); },
  show(selector) { this.css(selector, { display: '' }); },
};

// ============================
// Complex Media API Facade
// ============================
class MediaFacade {
  #stream = null;
  #recorder = null;
  #chunks = [];

  async startRecording() {
    this.#stream = await navigator.mediaDevices.getUserMedia({ audio: true });
    this.#recorder = new MediaRecorder(this.#stream);
    this.#chunks = [];
    this.#recorder.ondataavailable = (e) => this.#chunks.push(e.data);
    this.#recorder.start();
  }

  async stopRecording() {
    return new Promise(resolve => {
      this.#recorder.onstop = () => {
        const blob = new Blob(this.#chunks, { type: 'audio/webm' });
        this.#stream.getTracks().forEach(t => t.stop());
        resolve({ blob, url: URL.createObjectURL(blob) });
      };
      this.#recorder.stop();
    });
  }

  play(url) { return new Audio(url).play(); }
}
```
