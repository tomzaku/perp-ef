Compare CommonJS (CJS) and ECMAScript Modules (ESM):

1. Syntax differences: require/module.exports vs import/export
2. Loading behavior: synchronous vs asynchronous
3. Binding type: value copy vs live bindings
4. Static vs dynamic structure
5. Tree-shaking implications
6. Interoperability between CJS and ESM

## Examples

**Input:** // CJS
const lib = require("./lib");
// ESM
import { fn } from "./lib.js";
**Output:** Both import the module, but with different semantics
*CJS copies the exported value at require time. ESM creates a live binding that reflects updates.*


## Solution

```js
// ============================================
// 1. CommonJS (CJS) Syntax
// ============================================

// --- math.cjs ---
// Exporting
const add = (a, b) => a + b;
const subtract = (a, b) => a - b;
const PI = 3.14159;

// Named exports
module.exports.add = add;
module.exports.subtract = subtract;
module.exports.PI = PI;

// Or export an object at once
module.exports = { add, subtract, PI };

// Default-like export
module.exports = function multiply(a, b) {
  return a * b;
};

// --- app.cjs ---
// Importing
const math = require('./math.cjs');           // entire module
const { add, PI } = require('./math.cjs');     // destructuring

// Dynamic/conditional require
if (process.env.NODE_ENV === 'development') {
  const debug = require('./debug.cjs');
}

// ============================================
// 2. ECMAScript Modules (ESM) Syntax
// ============================================

// --- math.mjs ---
// Named exports
export const add = (a, b) => a + b;
export const subtract = (a, b) => a - b;
export const PI = 3.14159;

// Default export
export default function multiply(a, b) {
  return a * b;
}

// Re-exporting
export { add as sum } from './math.mjs';
export * from './helpers.mjs';

// --- app.mjs ---
// Named imports
import { add, PI } from './math.mjs';

// Default import
import multiply from './math.mjs';

// Combined
import multiply2, { add as add2, PI as PI2 } from './math.mjs';

// Namespace import
import * as math from './math.mjs';
math.add(1, 2);

// Dynamic import (returns a Promise)
const module = await import('./math.mjs');

// ============================================
// 3. Key Differences
// ============================================

/*
  Feature          | CommonJS (CJS)        | ESM
  -----------------|-----------------------|------------------------
  Syntax           | require/module.exports| import/export
  Loading          | Synchronous           | Asynchronous
  Parsing          | Runtime               | Parse time (static)
  Bindings         | Value copy            | Live bindings (read-only)
  Top-level await  | No                    | Yes
  this at top      | module.exports        | undefined
  File extension   | .cjs or .js           | .mjs or .js (w/ type:module)
  Tree-shaking     | Difficult             | Built-in support
  Conditional      | Yes (runtime)         | Only via dynamic import()
  Browser support  | No (needs bundler)    | Native
*/

// ============================================
// 4. Live Bindings vs Value Copy
// ============================================

// --- CJS: Value is COPIED at require time ---
// counter.cjs
let count = 0;
function increment() { count++; }
module.exports = { count, increment };

// app.cjs
const counter = require('./counter.cjs');
console.log(counter.count); // 0
counter.increment();
console.log(counter.count); // 0 ← still 0! (copied value)

// --- ESM: Live binding (reference to variable) ---
// counter.mjs
export let count = 0;
export function increment() { count++; }

// app.mjs
import { count, increment } from './counter.mjs';
console.log(count); // 0
increment();
console.log(count); // 1 ← updated! (live binding)
// count = 5;       // TypeError: read-only binding

// ============================================
// 5. Tree-Shaking (Dead Code Elimination)
// ============================================

// ESM enables tree-shaking because imports/exports are STATIC
// (known at parse time, not runtime)

// utils.mjs
export function usedFunction() { return 'I am used'; }
export function unusedFunction() { return 'I am dead code'; }

// app.mjs
import { usedFunction } from './utils.mjs';
// Bundler can remove unusedFunction from the final bundle!

// CJS CANNOT be tree-shaken reliably because:
// - require() is dynamic (can be conditional, in loops, etc.)
// - module.exports can be reassigned at any time
// - The bundler cannot statically analyze what's used

// ============================================
// 6. Interoperability
// ============================================

// ESM can import CJS (usually works)
import cjsModule from './module.cjs'; // gets module.exports as default

// CJS cannot use static import for ESM
// const esm = require('./module.mjs'); // ERROR!

// CJS can use dynamic import() for ESM
async function loadESM() {
  const esm = await import('./module.mjs');
  return esm.default;
}

// Package.json configuration
/*
{
  "type": "module",         // all .js files treated as ESM
  // or
  "type": "commonjs",       // all .js files treated as CJS (default)

  // Dual package support:
  "exports": {
    "import": "./dist/index.mjs",
    "require": "./dist/index.cjs"
  }
}
*/

// ============================================
// 7. Dynamic Import (Both Systems)
// ============================================
// import() works in both CJS and ESM and returns a Promise

// Code splitting / lazy loading
async function loadFeature(featureName) {
  const module = await import(`./features/${featureName}.js`);
  return module.default;
}

// Conditional loading
const adapter = await import(
  process.platform === 'win32'
    ? './adapters/windows.js'
    : './adapters/unix.js'
);
```
