The Strategy Pattern defines a family of algorithms, encapsulates each one, and makes them interchangeable. It lets the algorithm vary independently from clients that use it.

**Frontend use cases:**
- Sorting strategies (by name, date, price, relevance)
- Authentication strategies (OAuth, JWT, API key, session)
- Validation strategies (different rules for different fields)
- Rendering strategies (table, grid, list view)
- Payment processing (credit card, PayPal, crypto)

**Key benefit:** Adding a new strategy requires zero changes to the context code. This is OCP in action.

## Solution

```js
// ============================
// Sorting Strategies
// ============================
const sortStrategies = {
  byName: (a, b) => a.name.localeCompare(b.name),
  byPrice: (a, b) => a.price - b.price,
  byPriceDesc: (a, b) => b.price - a.price,
  byDate: (a, b) => new Date(b.date) - new Date(a.date),
  byRating: (a, b) => b.rating - a.rating,
};

function sortProducts(products, strategyName) {
  const strategy = sortStrategies[strategyName];
  if (!strategy) throw new Error(`Unknown sort: ${strategyName}`);
  return [...products].sort(strategy);
}

const products = [
  { name: 'Laptop', price: 999, date: '2024-01-15', rating: 4.5 },
  { name: 'Phone', price: 699, date: '2024-03-20', rating: 4.8 },
  { name: 'Tablet', price: 499, date: '2024-02-10', rating: 4.2 },
];

console.log(sortProducts(products, 'byPrice').map(p => p.name));
// ['Tablet', 'Phone', 'Laptop']

// Add new strategies without changing existing code
sortStrategies.byNameDesc = (a, b) => b.name.localeCompare(a.name);

// ============================
// Authentication Strategies
// ============================
class AuthContext {
  setStrategy(strategy) { this.strategy = strategy; }
  async authenticate(request) {
    if (!this.strategy) throw new Error('No auth strategy set');
    return this.strategy.authenticate(request);
  }
}

const jwtStrategy = {
  async authenticate(req) {
    const token = req.headers?.authorization?.replace('Bearer ', '');
    if (!token) throw new Error('No token');
    console.log('JWT verified');
    return { userId: 1, method: 'jwt' };
  }
};

const apiKeyStrategy = {
  async authenticate(req) {
    const key = req.headers?.['x-api-key'];
    if (!key) throw new Error('No API key');
    console.log('API key validated');
    return { userId: 2, method: 'apiKey' };
  }
};

const auth = new AuthContext();
auth.setStrategy(jwtStrategy);
auth.authenticate({ headers: { authorization: 'Bearer abc123' } });

auth.setStrategy(apiKeyStrategy);
auth.authenticate({ headers: { 'x-api-key': 'key_xyz' } });

// ============================
// Form Validation Strategies
// ============================
const validationStrategies = {
  required: (value) =>
    value !== '' && value != null ? null : 'This field is required',
  email: (value) =>
    /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value) ? null : 'Invalid email',
  minLength: (min) => (value) =>
    value.length >= min ? null : `Must be at least ${min} chars`,
  maxLength: (max) => (value) =>
    value.length <= max ? null : `Must be at most ${max} chars`,
  pattern: (regex, msg) => (value) =>
    regex.test(value) ? null : msg,
};

function validateField(value, rules) {
  return rules.map(rule => rule(value)).filter(Boolean);
}

const emailRules = [
  validationStrategies.required,
  validationStrategies.email,
  validationStrategies.maxLength(255),
];

const passwordRules = [
  validationStrategies.required,
  validationStrategies.minLength(8),
  validationStrategies.pattern(/[A-Z]/, 'Must contain uppercase'),
  validationStrategies.pattern(/[0-9]/, 'Must contain a number'),
];

console.log(validateField('', emailRules));
// ['This field is required', 'Invalid email']

console.log(validateField('weak', passwordRules));
// ['Must be at least 8 chars', 'Must contain uppercase', 'Must contain a number']

console.log(validateField('Strong1Pass', passwordRules));
// []
```
