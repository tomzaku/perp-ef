The Builder Pattern separates the construction of a complex object from its representation, allowing the same construction process to create different representations. It's especially useful when an object has many optional parameters.

**Frontend use cases:**
- Query/URL builders
- HTTP request builders
- Configuration objects with many options
- Test data builders (creating test fixtures)

**Key advantage:** Method chaining provides a fluent, readable API. Instead of a constructor with 10 parameters, the builder guides you step by step.

## Solution

```js
// ============================
// Query Builder
// ============================
class QueryBuilder {
  #table = '';
  #conditions = [];
  #orderFields = [];
  #limitVal = null;
  #offsetVal = null;
  #selectedFields = ['*'];

  from(table) { this.#table = table; return this; }
  select(...fields) { this.#selectedFields = fields; return this; }

  where(field, operator, value) {
    this.#conditions.push({ field, operator, value });
    return this;
  }

  orderBy(field, direction = 'ASC') {
    this.#orderFields.push(`${field} ${direction}`);
    return this;
  }

  limit(n) { this.#limitVal = n; return this; }
  offset(n) { this.#offsetVal = n; return this; }

  build() {
    let query = `SELECT ${this.#selectedFields.join(', ')} FROM ${this.#table}`;
    if (this.#conditions.length) {
      const clauses = this.#conditions.map(c => `${c.field} ${c.operator} '${c.value}'`).join(' AND ');
      query += ` WHERE ${clauses}`;
    }
    if (this.#orderFields.length) query += ` ORDER BY ${this.#orderFields.join(', ')}`;
    if (this.#limitVal !== null) query += ` LIMIT ${this.#limitVal}`;
    if (this.#offsetVal !== null) query += ` OFFSET ${this.#offsetVal}`;
    return query;
  }
}

const query = new QueryBuilder()
  .from('users')
  .select('id', 'name', 'email')
  .where('age', '>=', 18)
  .where('active', '=', true)
  .orderBy('name', 'ASC')
  .limit(10)
  .offset(20)
  .build();

console.log(query);
// SELECT id, name, email FROM users WHERE age >= '18' AND active = 'true' ORDER BY name ASC LIMIT 10 OFFSET 20

// ============================
// HTTP Request Builder
// ============================
class RequestBuilder {
  #url = '';
  #method = 'GET';
  #headers = {};
  #body = null;
  #params = {};
  #timeout = 5000;

  setUrl(url) { this.#url = url; return this; }
  setMethod(m) { this.#method = m; return this; }
  addHeader(k, v) { this.#headers[k] = v; return this; }
  setBody(data) { this.#body = data; return this; }
  addParam(k, v) { this.#params[k] = v; return this; }
  setTimeout(ms) { this.#timeout = ms; return this; }

  build() {
    const qs = new URLSearchParams(this.#params).toString();
    const fullUrl = qs ? `${this.#url}?${qs}` : this.#url;
    const options = {
      method: this.#method,
      headers: this.#headers,
      signal: AbortSignal.timeout(this.#timeout),
    };
    if (this.#body) {
      options.body = JSON.stringify(this.#body);
      options.headers['Content-Type'] = 'application/json';
    }
    return { url: fullUrl, options };
  }

  async execute() {
    const { url, options } = this.build();
    return fetch(url, options).then(r => r.json());
  }
}

const request = new RequestBuilder()
  .setUrl('https://api.example.com/users')
  .setMethod('POST')
  .addHeader('Authorization', 'Bearer token123')
  .setBody({ name: 'Alice' })
  .addParam('notify', 'true')
  .setTimeout(10000)
  .build();

console.log(request.url); // https://api.example.com/users?notify=true

// ============================
// Test Data Builder
// ============================
class UserBuilder {
  #user = { id: 1, name: 'Default User', email: 'default@test.com', role: 'viewer', active: true };

  withId(id) { this.#user.id = id; return this; }
  withName(name) { this.#user.name = name; return this; }
  withEmail(email) { this.#user.email = email; return this; }
  asAdmin() { this.#user.role = 'admin'; return this; }
  asInactive() { this.#user.active = false; return this; }
  build() { return { ...this.#user }; }
}

const admin = new UserBuilder().withName('Admin').asAdmin().build();
const inactive = new UserBuilder().withName('Old User').asInactive().build();
console.log(admin);    // { id: 1, name: 'Admin', role: 'admin', active: true }
console.log(inactive); // { id: 1, name: 'Old User', role: 'viewer', active: false }
```
