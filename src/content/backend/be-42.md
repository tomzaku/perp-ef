```text
PART 1: POSTGRESQL — Transactional Core
═══════════════════════════════════════════════════════

WHY SQL: Orders, payments, and inventory need ACID
transactions, foreign key constraints, and complex JOINs.

-- Schema
CREATE TABLE users (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email VARCHAR(255) UNIQUE NOT NULL,
  name VARCHAR(255) NOT NULL,
  password_hash TEXT NOT NULL,
  created_at TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE orders (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id),
  status VARCHAR(20) DEFAULT 'pending',
  total_cents BIGINT NOT NULL,  -- Store money as cents!
  created_at TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE order_items (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  order_id UUID REFERENCES orders(id),
  product_id UUID NOT NULL,
  quantity INT NOT NULL,
  price_cents BIGINT NOT NULL  -- Price at time of purchase
);

CREATE TABLE payments (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  order_id UUID REFERENCES orders(id),
  stripe_payment_id TEXT UNIQUE,
  amount_cents BIGINT NOT NULL,
  status VARCHAR(20) DEFAULT 'pending',
  created_at TIMESTAMPTZ DEFAULT now()
);

-- Transaction: Place an order (must be atomic)
BEGIN;
  -- Reserve inventory
  UPDATE products
  SET stock = stock - 2
  WHERE id = 'prod-123' AND stock >= 2;

  -- Create order
  INSERT INTO orders (user_id, total_cents)
  VALUES ('user-456', 5998)
  RETURNING id;

  -- Create order items
  INSERT INTO order_items (order_id, product_id, quantity, price_cents)
  VALUES ('order-789', 'prod-123', 2, 2999);

  -- Record payment intent
  INSERT INTO payments (order_id, amount_cents, stripe_payment_id)
  VALUES ('order-789', 5998, 'pi_abc123');
COMMIT;

-- Complex report: Monthly revenue by category
SELECT
  p.category,
  DATE_TRUNC('month', o.created_at) AS month,
  SUM(oi.price_cents * oi.quantity) AS revenue_cents,
  COUNT(DISTINCT o.id) AS order_count
FROM orders o
JOIN order_items oi ON o.id = oi.order_id
JOIN products p ON oi.product_id = p.id
WHERE o.status = 'completed'
  AND o.created_at >= '2024-01-01'
GROUP BY p.category, DATE_TRUNC('month', o.created_at)
ORDER BY month, revenue_cents DESC;

PART 2: REDIS — Cache, Sessions & Real-Time
═══════════════════════════════════════════════════════

WHY REDIS: Sub-millisecond reads for hot data.
No need for ACID here — it's a cache layer.
```

```javascript
const Redis = require('ioredis');
const redis = new Redis();

// ─── Shopping Cart (Hash) ─────────────────────
// Cart is temporary, doesn't need ACID
// Redis Hash = perfect for field-level updates

await redis.hset('cart:user-456', {
  'prod-123': JSON.stringify({ qty: 2, price: 2999 }),
  'prod-789': JSON.stringify({ qty: 1, price: 4999 }),
});
await redis.expire('cart:user-456', 86400 * 7); // 7 day TTL

// Get entire cart
const cart = await redis.hgetall('cart:user-456');

// Update single item quantity
await redis.hset('cart:user-456',
  'prod-123', JSON.stringify({ qty: 3, price: 2999 })
);

// Remove item from cart
await redis.hdel('cart:user-456', 'prod-789');


// ─── Session Store ────────────────────────────
// Fast auth lookups, auto-expiry via TTL

await redis.set(
  'session:abc123',
  JSON.stringify({ userId: 'user-456', role: 'customer' }),
  'EX', 3600  // 1 hour TTL
);

const session = JSON.parse(await redis.get('session:abc123'));


// ─── Product Cache (Cache-Aside Pattern) ──────
// Read-heavy product pages cached from Postgres

async function getProduct(id) {
  // 1. Check cache
  const cached = await redis.get(`product:${id}`);
  if (cached) return JSON.parse(cached);

  // 2. Cache miss → query Postgres
  const product = await db.query(
    'SELECT * FROM products WHERE id = $1', [id]
  );

  // 3. Populate cache with TTL
  await redis.set(
    `product:${id}`,
    JSON.stringify(product),
    'EX', 300  // 5 minutes
  );
  return product;
}

// Invalidate on update
async function updateProduct(id, data) {
  await db.query('UPDATE products SET name=$1 WHERE id=$2', [data.name, id]);
  await redis.del(`product:${id}`);  // Invalidate cache
}


// ─── Rate Limiting (Sliding Window) ───────────

async function rateLimit(userId, limit = 100, windowSec = 60) {
  const key = `ratelimit:${userId}`;
  const now = Date.now();

  await redis
    .multi()
    .zadd(key, now, `${now}`)                    // Add current request
    .zremrangebyscore(key, 0, now - windowSec * 1000) // Remove old
    .zcard(key)                                  // Count in window
    .expire(key, windowSec)
    .exec();

  const count = await redis.zcard(key);
  return count <= limit;
}


// ─── Leaderboard (Sorted Set) ─────────────────

// Add scores
await redis.zadd('leaderboard:sellers', 15000, 'seller-A');
await redis.zadd('leaderboard:sellers', 22000, 'seller-B');
await redis.zadd('leaderboard:sellers', 18000, 'seller-C');

// Top 10 sellers by revenue
const top10 = await redis.zrevrange('leaderboard:sellers', 0, 9, 'WITHSCORES');
// → ['seller-B', '22000', 'seller-C', '18000', 'seller-A', '15000']
```

```javascript
  category: "electronics",
  "attributes.ram_gb": { $gte: 16 },
  price_cents: { $lt: 200000 }

// Query: Aggregate best-selling categories
db.products.aggregate([
  { $group: {
    _id: "$category",
    avgPrice: { $avg: "$price_cents" },
    totalProducts: { $sum: 1 },
    avgRating: { $avg: "$ratings.average" }
  }},
  { $sort: { totalProducts: -1 } }
]);
```

```javascript
// Index a product into Elasticsearch
await elastic.index({
  index: 'products',
  id: 'prod-123',
  body: {
    name: "MacBook Pro 16",
    description: "Powerful laptop for professionals...",
    category: "electronics",
    brand: "Apple",
    price_cents: 249900,
    tags: ["laptop", "apple", "professional"],
    rating: 4.7
  }
});

// Search: "macbok pro" (typo-tolerant)
const results = await elastic.search({
  index: 'products',
  body: {
    query: {
      bool: {
        must: [
          { multi_match: {
            query: "macbok pro",     // User typo!
            fields: ["name^3", "description", "tags"],
            fuzziness: "AUTO"        // Handles typos
          }}
        ],
        filter: [
          { range: { price_cents: { lte: 300000 } } },
          { term: { category: "electronics" } }
        ]
      }
    },
    aggs: {                          // Faceted filters
      brands: { terms: { field: "brand.keyword" } },
      price_ranges: {
        range: {
          field: "price_cents",
          ranges: [
            { to: 50000, key: "Under $500" },
            { from: 50000, to: 150000, key: "$500-$1500" },
            { from: 150000, key: "Over $1500" }
          ]
        }
      }
    }
  }
});

Options:
1. Change Data Capture (Debezium) — reads Postgres WAL
2. Application-level events — publish on write
3. Periodic sync jobs — simple but introduces lag
```

## Explanation

**Why not just use one database for everything?**

| Scenario | Best Tool | Why not Postgres? |
|---|---|---|
| Shopping cart | Redis | Temporary data, needs sub-ms reads, auto-expiry via TTL |
| Product catalog | MongoDB | Each product type has different fields — flexible schema avoids 50-column sparse tables |
| Full-text search | Elasticsearch | Fuzzy matching, relevance scoring, faceted filters at scale |
| Order processing | PostgreSQL | Needs ACID transactions, foreign keys, complex JOINs for reporting |
| Session storage | Redis | Millions of reads/sec, built-in expiry, no JOINs needed |
| Analytics | ClickHouse | Columnar storage, 100x faster for aggregation queries on billions of rows |

**The sync problem** is the biggest challenge in polyglot persistence. When a product price changes in Postgres, Redis cache, Elasticsearch index, and MongoDB catalog all need updating. Use Change Data Capture (Debezium reads the Postgres WAL) or application-level events via a message queue (Kafka) to keep stores in sync.

## ELI5

Imagine running a restaurant. You use a **cash register** (PostgreSQL) for all the money — receipts, payments, accounting. You have a **chalkboard** (Redis) by the door showing today's specials — quick to read, easy to erase. The **menu** (MongoDB) has different sections with different layouts — appetizers have photos, drinks have sizes, desserts have allergen info. And there's a **phone book** (Elasticsearch) so customers can search for dishes even if they spell "bruschetta" wrong. Each tool does one thing really well.
