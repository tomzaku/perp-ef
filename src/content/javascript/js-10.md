Explain WeakMap and WeakSet in JavaScript:

1. How they differ from Map and Set
2. Garbage collection behavior (weak references)
3. Use cases: private data, DOM metadata, caching, object tagging
4. Limitations: no iteration, no size property
5. WeakRef and FinalizationRegistry (bonus)

## Examples

**Input:** const wm = new WeakMap(); let obj = {}; wm.set(obj, "data"); obj = null;
**Output:** The WeakMap entry is eligible for garbage collection
*WeakMap holds weak references to keys. When no other references to the key exist, the entry can be garbage collected.*


## Solution

```js
// ============================================
// 1. WeakMap vs Map
// ============================================

// Map: Strong references — keys prevent garbage collection
const map = new Map();
let obj1 = { name: 'Alice' };
map.set(obj1, 'metadata');
obj1 = null;
// obj1 is still referenced by map — NOT garbage collected
console.log(map.size); // 1 (entry persists)

// WeakMap: Weak references — keys DO NOT prevent GC
const weakMap = new WeakMap();
let obj2 = { name: 'Bob' };
weakMap.set(obj2, 'metadata');
obj2 = null;
// obj2 has no strong references left — eligible for GC
// The WeakMap entry is automatically cleaned up

// Key restrictions: WeakMap keys MUST be objects (or symbols)
// weakMap.set('string', 'value'); // TypeError!
// weakMap.set(42, 'value');       // TypeError!

// No iteration, no .size, no .keys(), no .values(), no .entries()
// Only: .get(), .set(), .has(), .delete()

// ============================================
// 2. Use Case: Private Instance Data
// ============================================
const privateData = new WeakMap();

class User {
  constructor(name, password) {
    this.name = name;
    // Store password privately — not accessible from outside
    privateData.set(this, { password, loginAttempts: 0 });
  }

  authenticate(password) {
    const data = privateData.get(this);
    if (data.password === password) {
      data.loginAttempts = 0;
      return true;
    }
    data.loginAttempts++;
    if (data.loginAttempts >= 3) {
      throw new Error('Account locked');
    }
    return false;
  }

  // When User instance is garbage collected,
  // the private data is automatically cleaned up too!
}

const user = new User('Alice', 'secret123');
console.log(user.authenticate('secret123')); // true
console.log(user.name);     // "Alice"
// user.password => undefined (truly private!)
// privateData is not exposed

// ============================================
// 3. Use Case: DOM Element Metadata
// ============================================
const elementData = new WeakMap();

function trackElement(element) {
  elementData.set(element, {
    clickCount: 0,
    created: Date.now(),
    handlers: []
  });
}

function recordClick(element) {
  const data = elementData.get(element);
  if (data) {
    data.clickCount++;
  }
}

// When the DOM element is removed and garbage collected,
// the metadata in the WeakMap is automatically freed.
// No memory leaks!

// ============================================
// 4. Use Case: Memoization Cache
// ============================================
const cache = new WeakMap();

function expensiveComputation(obj) {
  if (cache.has(obj)) {
    console.log('Cache hit');
    return cache.get(obj);
  }

  console.log('Computing...');
  const result = {
    ...obj,
    computed: Object.keys(obj).length * 42
  };

  cache.set(obj, result);
  return result;
}

let data = { a: 1, b: 2 };
expensiveComputation(data); // "Computing..."
expensiveComputation(data); // "Cache hit"
data = null;
// Cache entry is automatically garbage collected

// ============================================
// 5. WeakSet: Object Tagging
// ============================================
const visited = new WeakSet();

function processNode(node) {
  if (visited.has(node)) {
    console.log('Already processed:', node.id);
    return;
  }

  visited.add(node);
  console.log('Processing:', node.id);
  // Do work...
}

// Useful for: cycle detection in graph traversal
function hasCycle(obj, seen = new WeakSet()) {
  if (obj === null || typeof obj !== 'object') return false;
  if (seen.has(obj)) return true;

  seen.add(obj);
  for (const value of Object.values(obj)) {
    if (hasCycle(value, seen)) return true;
  }
  return false;
}

const circular = { a: 1 };
circular.self = circular;
console.log(hasCycle(circular)); // true
console.log(hasCycle({ a: 1, b: { c: 2 } })); // false

// WeakSet for branding / type checking
const validatedInputs = new WeakSet();

function validateInput(input) {
  // ... validation logic ...
  if (typeof input === 'object' && input !== null) {
    validatedInputs.add(input);
  }
  return input;
}

function processInput(input) {
  if (!validatedInputs.has(input)) {
    throw new Error('Input must be validated first');
  }
  // Process safely...
}

// 6. WeakRef and FinalizationRegistry
// ============================================
// WeakRef: holds a weak reference to an object
// FinalizationRegistry: callback when object is GC'd

class ImageCache {
  #cache = new Map();
  #registry = new FinalizationRegistry((key) => {
    console.log(`Image "${key}" was garbage collected`);
    this.#cache.delete(key);
  });

  set(key, imageData) {
    const ref = new WeakRef(imageData);
    this.#cache.set(key, ref);
    this.#registry.register(imageData, key);
  }

  get(key) {
    const ref = this.#cache.get(key);
    if (!ref) return undefined;

    const value = ref.deref(); // may return undefined if GC'd
    if (value === undefined) {
      this.#cache.delete(key); // clean up stale ref
    }
    return value;
  }

  get size() {
    return this.#cache.size;
  }
}

// Usage:
const imgCache = new ImageCache();
let largeImage = { data: new ArrayBuffer(1024 * 1024), name: 'photo.jpg' };
imgCache.set('photo', largeImage);
console.log(imgCache.get('photo')); // { data: ..., name: 'photo.jpg' }
largeImage = null;
// Eventually after GC: "Image "photo" was garbage collected"
```
