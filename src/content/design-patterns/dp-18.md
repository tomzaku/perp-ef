The Command Pattern encapsulates a request as an object, allowing you to parameterize operations, queue them, log them, and support undo/redo.

**Frontend use cases:**
- Text editor undo/redo
- Action history and replay
- Macro recording (execute a sequence of commands)
- Task queues and job scheduling
- Transaction-like operations (execute all or rollback)

**Key components:**
- **Command:** object with execute() and undo() methods
- **Invoker:** triggers commands (e.g., a toolbar button)
- **Receiver:** the object being acted upon (e.g., the document)
- **History:** stack of executed commands for undo/redo

## Solution

```js
// ============================
// Text Editor with Undo/Redo
// ============================
class TextEditor {
  constructor() {
    this.content = '';
    this.history = [];
    this.redoStack = [];
  }

  execute(command) {
    command.execute();
    this.history.push(command);
    this.redoStack = [];
  }

  undo() {
    const cmd = this.history.pop();
    if (cmd) { cmd.undo(); this.redoStack.push(cmd); }
  }

  redo() {
    const cmd = this.redoStack.pop();
    if (cmd) { cmd.execute(); this.history.push(cmd); }
  }

  getContent() { return this.content; }
}

class InsertTextCommand {
  constructor(editor, text, position) {
    this.editor = editor;
    this.text = text;
    this.position = position;
  }
  execute() {
    const before = this.editor.content.slice(0, this.position);
    const after = this.editor.content.slice(this.position);
    this.editor.content = before + this.text + after;
  }
  undo() {
    const before = this.editor.content.slice(0, this.position);
    const after = this.editor.content.slice(this.position + this.text.length);
    this.editor.content = before + after;
  }
}

class DeleteTextCommand {
  constructor(editor, position, length) {
    this.editor = editor;
    this.position = position;
    this.length = length;
    this.deletedText = '';
  }
  execute() {
    this.deletedText = this.editor.content.slice(this.position, this.position + this.length);
    const before = this.editor.content.slice(0, this.position);
    const after = this.editor.content.slice(this.position + this.length);
    this.editor.content = before + after;
  }
  undo() {
    const before = this.editor.content.slice(0, this.position);
    const after = this.editor.content.slice(this.position);
    this.editor.content = before + this.deletedText + after;
  }
}

class ReplaceTextCommand {
  constructor(editor, search, replacement) {
    this.editor = editor;
    this.search = search;
    this.replacement = replacement;
    this.previousContent = '';
  }
  execute() {
    this.previousContent = this.editor.content;
    this.editor.content = this.editor.content.replace(this.search, this.replacement);
  }
  undo() {
    this.editor.content = this.previousContent;
  }
}

// Usage
const editor = new TextEditor();

editor.execute(new InsertTextCommand(editor, 'Hello', 0));
console.log(editor.getContent()); // "Hello"

editor.execute(new InsertTextCommand(editor, ' World', 5));
console.log(editor.getContent()); // "Hello World"

editor.execute(new ReplaceTextCommand(editor, 'World', 'JavaScript'));
console.log(editor.getContent()); // "Hello JavaScript"

editor.undo();
console.log(editor.getContent()); // "Hello World"

editor.undo();
console.log(editor.getContent()); // "Hello"

editor.redo();
console.log(editor.getContent()); // "Hello World"

// ============================
// Command Queue / Macro
// ============================
class CommandQueue {
  constructor() { this.queue = []; }

  add(command) { this.queue.push(command); return this; }

  executeAll() {
    const results = [];
    for (const cmd of this.queue) results.push(cmd.execute());
    this.queue = [];
    return results;
  }
}

const macro = new CommandQueue();
macro
  .add(new InsertTextCommand(editor, '!', editor.getContent().length))
  .add(new ReplaceTextCommand(editor, 'Hello', 'Hi'));
macro.executeAll();
console.log(editor.getContent()); // "Hi World!"
```
