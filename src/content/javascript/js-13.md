Explain the Temporal Dead Zone (TDZ) and variable hoisting in JavaScript:

1. How var, let, and const are hoisted differently
2. What the TDZ is and why it exists
3. TDZ with function parameters, class declarations, and more
4. Practical implications and common gotchas

## Examples

**Input:** console.log(x); let x = 5;
**Output:** ReferenceError: Cannot access "x" before initialization
*let is hoisted but not initialized. Accessing it before declaration throws due to TDZ.*

**Input:** console.log(y); var y = 5;
**Output:** undefined
*var is hoisted AND initialized to undefined. No TDZ.*


## Solution

```js
// ============================================
// 1. var Hoisting (No TDZ)
// ============================================
// var declarations are hoisted to the top of their function scope
// and initialized with undefined.

console.log(a); // undefined (hoisted, initialized to undefined)
var a = 10;
console.log(a); // 10

// The engine sees it as:
// var a = undefined;  (hoisted)
// console.log(a);     // undefined
// a = 10;
// console.log(a);     // 10

// var is FUNCTION-scoped, not block-scoped
function example() {
  console.log(x); // undefined (hoisted to function top)
  if (true) {
    var x = 5;    // hoisted to function scope, NOT block
  }
  console.log(x); // 5
}

// ============================================
// 2. let and const Hoisting (TDZ)
// ============================================
// let and const ARE hoisted, but NOT initialized.
// The time between hoisting and declaration is the TDZ.

// console.log(b); // ReferenceError: Cannot access 'b' before initialization
let b = 20;

// console.log(c); // ReferenceError: Cannot access 'c' before initialization
const c = 30;

// The TDZ exists from the start of the block to the declaration:
{
  // TDZ for 'name' starts here
  // console.log(name); // ReferenceError    TDZ
  // const temp = name; // ReferenceError
  let name = 'Alice'; // TDZ ends
  console.log(name);  // "Alice" (safe after declaration)
}

// ============================================
// 3. TDZ is Temporal, Not Spatial
// ============================================
// TDZ is about TIME of access, not POSITION in code

function works() {
  // This function is defined BEFORE 'value', but
  // it's CALLED AFTER 'value' is initialized — so it works!
  function inner() {
    return value; // references 'value' before its textual position
  }

  let value = 42;
  return inner(); // 42 — called after initialization, no TDZ error
}

function fails() {
  function inner() {
    return value;
  }

  inner(); // ReferenceError! Called before 'value' is initialized
  let value = 42;
}

// ============================================
// 4. var vs let vs const Comparison
// ============================================

/*
  Feature        | var          | let          | const
  ---------------|--------------|--------------|-------------
  Scope          | Function     | Block        | Block
  Hoisting       | Yes + init   | Yes, no init | Yes, no init
  TDZ            | No           | Yes          | Yes
  Redeclaration  | Yes          | No           | No
  Reassignment   | Yes          | Yes          | No*
  Global object  | Yes (window) | No           | No

  * const prevents reassignment of the binding, not mutation of the value.
*/

// const with objects — the reference is constant, not the value
const person = { name: 'Alice' };
person.name = 'Bob';       // OK (mutating the object)
person.age = 30;           // OK (adding property)
// person = { name: 'Carol' }; // TypeError: Assignment to constant variable

// const with arrays
const arr = [1, 2, 3];
arr.push(4);     // OK
arr[0] = 99;     // OK
// arr = [5, 6]; // TypeError

// Use Object.freeze for true immutability (shallow)
const frozen = Object.freeze({ name: 'Alice', nested: { x: 1 } });
// frozen.name = 'Bob'; // silently fails (or throws in strict mode)
frozen.nested.x = 999;  // WORKS! freeze is shallow

// ============================================
// 5. TDZ in Other Contexts
// ============================================

// Default parameters
// function greet(name = greeting, greeting = 'Hello') {
//   return `${greeting} ${name}`;
// }
// greet(); // ReferenceError: 'greeting' is in TDZ when 'name' evaluates

// Class declarations have TDZ too
// const instance = new MyClass(); // ReferenceError
// class MyClass {}

// But class EXPRESSIONS assigned to var:
// var instance2 = new MyClass2(); // TypeError: MyClass2 is not a constructor
// var MyClass2 = class {};        // var hoists as undefined

// switch/case shares scope
// switch (x) {
//   case 0:
//     let name = 'zero'; // declared in switch block
//     break;
//   case 1:
//     let name = 'one';  // SyntaxError: 'name' already declared
//     break;
// }

// Fix: use blocks in case clauses
switch (1) {
  case 0: {
    let name = 'zero';
    break;
  }
  case 1: {
    let name = 'one'; // OK, separate block scope
    console.log(name);
    break;
  }
}

// ============================================
// 6. Function Hoisting
// ============================================

// Function declarations are FULLY hoisted (body + name)
console.log(greet('World')); // "Hello World"
function greet(name) {
  return `Hello ${name}`;
}

// Function expressions follow their variable's hoisting rules
// console.log(sayHi); // undefined (var hoisted)
// sayHi('World');     // TypeError: sayHi is not a function
var sayHi = function (name) {
  return `Hi ${name}`;
};

// Arrow functions also follow variable rules
// console.log(arrow); // ReferenceError (let TDZ)
// arrow();            // ReferenceError
let arrow = () => 'I am an arrow';

// ============================================
// 7. Best Practices
// ============================================
// 1. Use const by default
// 2. Use let when you need to reassign
// 3. Never use var (function scope + no TDZ = bugs)
// 4. Declare variables at the top of their scope
// 5. Always initialize const at declaration (required)
```
