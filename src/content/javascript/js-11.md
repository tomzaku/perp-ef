Explain JavaScript Proxy and Reflect APIs:

1. Proxy traps and handler methods
2. Reflect API and why it exists
3. Practical use cases: validation, logging, observable objects
4. Negative indexing for arrays
5. Creating a reactive system (simplified Vue-style reactivity)

## Examples

**Input:** const proxy = new Proxy(target, { get(target, prop) { ... } }); proxy.name
**Output:** Intercepted property access
*Proxy intercepts fundamental operations on objects via handler traps.*


## Solution

```js
// ============================================
// 1. Proxy Basics
// ============================================
const handler = {
  get(target, property, receiver) {
    console.log(`GET ${String(property)}`);
    return Reflect.get(target, property, receiver);
  },
  set(target, property, value, receiver) {
    console.log(`SET ${String(property)} = ${value}`);
    return Reflect.set(target, property, value, receiver);
  },
  has(target, property) {
    console.log(`HAS ${String(property)}`);
    return Reflect.has(target, property);
  },
  deleteProperty(target, property) {
    console.log(`DELETE ${String(property)}`);
    return Reflect.deleteProperty(target, property);
  }
};

const obj = new Proxy({ name: 'Alice', age: 30 }, handler);
obj.name;          // GET name => "Alice"
obj.age = 31;      // SET age = 31
'name' in obj;     // HAS name => true
delete obj.age;    // DELETE age

// ============================================
// 2. Validation Proxy
// ============================================
function createValidatedObject(schema) {
  return new Proxy({}, {
    set(target, property, value) {
      const validator = schema[property];
      if (!validator) {
        throw new Error(`Unknown property: ${String(property)}`);
      }

      const error = validator(value);
      if (error) {
        throw new TypeError(`Invalid value for ${String(property)}: ${error}`);
      }

      return Reflect.set(target, property, value);
    },

    get(target, property) {
      if (!(property in target) && property in schema) {
        throw new Error(`Property ${String(property)} has not been set yet`);
      }
      return Reflect.get(target, property);
    }
  });
}

const user = createValidatedObject({
  name: (v) =>
    typeof v !== 'string' ? 'must be a string' :
    v.length < 1 ? 'must not be empty' : null,
  age: (v) =>
    typeof v !== 'number' ? 'must be a number' :
    v < 0 || v > 150 ? 'must be between 0 and 150' : null,
  email: (v) =>
    typeof v !== 'string' ? 'must be a string' :
    !v.includes('@') ? 'must be a valid email' : null
});

user.name = 'Alice';   // OK
user.age = 30;          // OK
// user.age = -5;       // TypeError: must be between 0 and 150
// user.foo = 'bar';    // Error: Unknown property: foo

// ============================================
// 3. Logging / Profiling Proxy
// ============================================
function withLogging(obj, label = 'Object') {
  return new Proxy(obj, {
    get(target, property, receiver) {
      const value = Reflect.get(target, property, receiver);
      if (typeof value === 'function') {
        return function (...args) {
          const start = performance.now();
          console.log(`${label}.${String(property)}(${args.map(a => JSON.stringify(a)).join(', ')})`);
          const result = value.apply(this === receiver ? target : this, args);
          const duration = (performance.now() - start).toFixed(2);
          console.log(`  => ${JSON.stringify(result)} (${duration}ms)`);
          return result;
        };
      }
      return value;
    }
  });
}

const math = withLogging(Math, 'Math');
math.max(1, 5, 3);    // Math.max(1, 5, 3) => 5 (0.01ms)
math.floor(4.7);       // Math.floor(4.7) => 4 (0.00ms)

// ============================================
// 4. Negative Array Indexing
// ============================================
function createNegativeArray(arr) {
  return new Proxy(arr, {
    get(target, property, receiver) {
      const index = Number(property);
      if (!Number.isNaN(index) && index < 0) {
        property = String(target.length + index);
      }
      return Reflect.get(target, property, receiver);
    },
    set(target, property, value) {
      const index = Number(property);
      if (!Number.isNaN(index) && index < 0) {
        property = String(target.length + index);
      }
      return Reflect.set(target, property, value);
    }
  });
}

const arr = createNegativeArray([1, 2, 3, 4, 5]);
console.log(arr[-1]); // 5 (last element)
console.log(arr[-2]); // 4 (second to last)
arr[-1] = 99;
console.log(arr);      // [1, 2, 3, 4, 99]

// ============================================
// 5. Observable Object (Reactive System)
// ============================================
function observable(target) {
  const listeners = new Map();

  function on(property, callback) {
    if (!listeners.has(property)) {
      listeners.set(property, new Set());
    }
    listeners.get(property).add(callback);
    return () => listeners.get(property).delete(callback); // unsubscribe
  }

  const proxy = new Proxy(target, {
    set(target, property, value, receiver) {
      const oldValue = target[property];
      const result = Reflect.set(target, property, value, receiver);

      if (oldValue !== value && listeners.has(property)) {
        listeners.get(property).forEach((callback) => {
          callback(value, oldValue, property);
        });
      }

      return result;
    }
  });

  proxy.__on = on;
  return proxy;
}

const state = observable({ count: 0, name: 'test' });

const unsubscribe = state.__on('count', (newVal, oldVal) => {
  console.log(`count changed: ${oldVal} -> ${newVal}`);
});

state.count = 1;  // "count changed: 0 -> 1"
state.count = 2;  // "count changed: 1 -> 2"
unsubscribe();
state.count = 3;  // (no log — unsubscribed)

// ============================================
// 6. Revocable Proxy (Access Control)
// ============================================
function createAccessToken(data, ttlMs) {
  const { proxy, revoke } = Proxy.revocable(data, {
    get(target, property) {
      return Reflect.get(target, property);
    }
  });

  // Auto-revoke after TTL
  setTimeout(() => {
    revoke();
    console.log('Access token expired');
  }, ttlMs);

  return proxy;
}

const secret = createAccessToken({ apiKey: 'abc123', secret: 'xyz' }, 5000);
console.log(secret.apiKey); // "abc123"
// After 5 seconds:
// secret.apiKey => TypeError: Cannot perform 'get' on a proxy that has been revoked

// ============================================
// 7. Why Reflect?
// ============================================
// Reflect provides default behavior for each Proxy trap:
// - Reflect.get, Reflect.set, Reflect.has, etc.
// - Returns boolean for success/failure instead of throwing
// - Proper handling of receiver for inheritance
// - Cleaner than Object.xxx counterparts

// Without Reflect (error-prone):
// target[property] = value; // doesn't trigger setters on prototype

// With Reflect (correct):
// Reflect.set(target, property, value, receiver); // handles inheritance
```
