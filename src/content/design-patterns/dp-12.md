The Adapter Pattern converts the interface of a class into another interface that clients expect. It lets classes work together that couldn't otherwise because of incompatible interfaces.

**Frontend use cases:**
- Wrapping a third-party API to match your app's expected interface
- Normalizing data from different API versions (v1 vs v2)
- Adapting a legacy component to work with a new system
- Unifying different storage backends behind one API

**Think of it like a power adapter:** Your laptop expects USB-C, but the wall outlet is different. The adapter bridges the gap without modifying either side.

## Solution

```js
// ============================
// API Adapter: normalize different API responses
// ============================
class OldUserApi {
  async getUser(id) {
    return { user_id: id, first_name: 'John', last_name: 'Doe', email_address: 'john@example.com', is_active: 1 };
  }
}

class NewUserApi {
  async getUser(id) {
    return { id, firstName: 'John', lastName: 'Doe', email: 'john@example.com', active: true };
  }
}

// Adapter: makes old API look like new API
class OldApiAdapter {
  constructor(oldApi) { this.oldApi = oldApi; }

  async getUser(id) {
    const old = await this.oldApi.getUser(id);
    return {
      id: old.user_id,
      firstName: old.first_name,
      lastName: old.last_name,
      email: old.email_address,
      active: Boolean(old.is_active),
    };
  }
}

// Both APIs work interchangeably
async function displayUser(userApi, id) {
  const user = await userApi.getUser(id);
  console.log(`${user.firstName} ${user.lastName} (${user.email})`);
}

displayUser(new OldApiAdapter(new OldUserApi()), 1);
displayUser(new NewUserApi(), 1);

// ============================
// Storage Adapter: unify different backends
// ============================
class LocalStorageAdapter {
  get(key) {
    const item = localStorage.getItem(key);
    return item ? JSON.parse(item) : null;
  }
  set(key, value) { localStorage.setItem(key, JSON.stringify(value)); }
  remove(key) { localStorage.removeItem(key); }
}

class InMemoryStorageAdapter {
  constructor() { this.store = new Map(); }
  get(key) { return this.store.get(key) || null; }
  set(key, value) { this.store.set(key, value); }
  remove(key) { this.store.delete(key); }
}

class CookieStorageAdapter {
  get(key) {
    const match = document.cookie.match(new RegExp(`(^| )${key}=([^;]+)`));
    return match ? JSON.parse(decodeURIComponent(match[2])) : null;
  }
  set(key, value, days = 7) {
    const expires = new Date(Date.now() + days * 864e5).toUTCString();
    document.cookie = `${key}=${encodeURIComponent(JSON.stringify(value))};expires=${expires};path=/`;
  }
  remove(key) {
    document.cookie = `${key}=;expires=Thu, 01 Jan 1970 00:00:00 GMT;path=/`;
  }
}

// App code uses one interface regardless of backend
function createPrefsManager(storage) {
  return {
    getTheme() { return storage.get('theme') || 'light'; },
    setTheme(theme) { storage.set('theme', theme); },
  };
}

const prefs = createPrefsManager(new InMemoryStorageAdapter());
prefs.setTheme('dark');
console.log(prefs.getTheme()); // 'dark'
```
