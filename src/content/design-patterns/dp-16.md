The Observer Pattern defines a one-to-many dependency between objects so that when one object (the subject) changes state, all its dependents (observers) are notified automatically.

**This is one of the most important patterns in frontend development:**
- DOM event listeners are observers
- React's setState triggers re-renders (observers)
- Redux store subscribers
- RxJS Observables
- Node.js EventEmitter

**Pub/Sub** is a variation where publishers and subscribers are decoupled through a message broker, so they don't know about each other directly.

Implement a custom EventEmitter from scratch — a very common interview question.

## Solution

```js
// ============================
// EventEmitter from scratch
// ============================
class EventEmitter {
  constructor() {
    this.listeners = new Map();
  }

  on(event, callback) {
    if (!this.listeners.has(event)) this.listeners.set(event, []);
    this.listeners.get(event).push(callback);
    return () => this.off(event, callback); // Return unsubscribe fn
  }

  off(event, callback) {
    if (!this.listeners.has(event)) return;
    const cbs = this.listeners.get(event);
    const idx = cbs.indexOf(callback);
    if (idx > -1) cbs.splice(idx, 1);
    if (cbs.length === 0) this.listeners.delete(event);
  }

  emit(event, ...args) {
    if (!this.listeners.has(event)) return;
    [...this.listeners.get(event)].forEach(cb => cb(...args));
  }

  once(event, callback) {
    const wrapper = (...args) => {
      callback(...args);
      this.off(event, wrapper);
    };
    this.on(event, wrapper);
  }

  removeAllListeners(event) {
    if (event) this.listeners.delete(event);
    else this.listeners.clear();
  }

  listenerCount(event) {
    return this.listeners.has(event) ? this.listeners.get(event).length : 0;
  }
}

// Usage
const emitter = new EventEmitter();

const unsub = emitter.on('userLogin', (user) => {
  console.log(`Welcome, ${user.name}!`);
});

emitter.on('userLogin', (user) => {
  console.log(`Log: ${user.name} logged in`);
});

emitter.once('userLogin', () => {
  console.log('First login bonus!');
});

emitter.emit('userLogin', { name: 'Alice' });
// Welcome, Alice!
// Log: Alice logged in
// First login bonus!

emitter.emit('userLogin', { name: 'Bob' });
// Welcome, Bob!
// Log: Bob logged in
// (no bonus — once listener removed)

unsub(); // Unsubscribe first listener

// ============================
// Pub/Sub Event Bus
// ============================
class EventBus {
  constructor() { this.emitter = new EventEmitter(); }

  publish(channel, data) {
    this.emitter.emit(channel, { channel, data, timestamp: Date.now() });
  }

  subscribe(channel, handler) {
    return this.emitter.on(channel, handler);
  }
}

const bus = new EventBus();

// Module A: Cart
bus.subscribe('item:added', ({ data }) => {
  console.log(`Cart: added ${data.name}`);
});

// Module B: Analytics (decoupled from cart)
bus.subscribe('item:added', ({ data }) => {
  console.log(`Analytics: track item ${data.id}`);
});

// Module C: triggers event, doesn't know about A or B
bus.publish('item:added', { id: 42, name: 'Widget', price: 9.99 });
// Cart: added Widget
// Analytics: track item 42
```
