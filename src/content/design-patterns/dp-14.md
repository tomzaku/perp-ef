The Proxy Pattern provides a surrogate or placeholder for another object to control access to it. Unlike the Decorator (which adds behavior), a Proxy controls access — intercepting operations and adding cross-cutting concerns.

**JavaScript has a built-in Proxy object** that intercepts fundamental operations (get, set, delete, function calls, etc.).

**Frontend use cases:**
- Validation proxy (reject invalid property assignments)
- Caching proxy (return cached results for repeated calls)
- Logging/debugging proxy (log all property accesses)
- Lazy initialization (defer expensive creation until first access)
- Reactive state (Vue 3 uses Proxy for reactivity)

## Solution

```js
// ============================
// Validation Proxy
// ============================
function createValidatedUser(user) {
  return new Proxy(user, {
    set(target, prop, value) {
      if (prop === 'age') {
        if (typeof value !== 'number' || value < 0 || value > 150)
          throw new TypeError('Age must be a number between 0 and 150');
      }
      if (prop === 'email') {
        if (typeof value !== 'string' || !value.includes('@'))
          throw new TypeError('Invalid email address');
      }
      if (prop === 'name') {
        if (typeof value !== 'string' || value.length < 1)
          throw new TypeError('Name must be a non-empty string');
      }
      target[prop] = value;
      return true;
    }
  });
}

const user = createValidatedUser({ name: 'Alice', age: 25, email: 'alice@test.com' });
user.name = 'Bob'; // OK
user.age = 30;     // OK
// user.age = -5;  // TypeError!

// ============================
// Caching Proxy for functions
// ============================
function createCachingProxy(fn, ttlMs = 60000) {
  const cache = new Map();
  return new Proxy(fn, {
    apply(target, thisArg, args) {
      const key = JSON.stringify(args);
      const cached = cache.get(key);
      if (cached && Date.now() - cached.ts < ttlMs) {
        console.log('Cache hit');
        return Promise.resolve(cached.data);
      }
      return target.apply(thisArg, args).then(data => {
        cache.set(key, { data, ts: Date.now() });
        return data;
      });
    }
  });
}

// ============================
// Logging Proxy
// ============================
function createLoggingProxy(obj, name = 'Object') {
  return new Proxy(obj, {
    get(target, prop) {
      console.log(`[GET] ${name}.${String(prop)}`);
      const value = target[prop];
      return typeof value === 'function' ? value.bind(target) : value;
    },
    set(target, prop, value) {
      console.log(`[SET] ${name}.${String(prop)} = ${JSON.stringify(value)}`);
      target[prop] = value;
      return true;
    },
    deleteProperty(target, prop) {
      console.log(`[DELETE] ${name}.${String(prop)}`);
      delete target[prop];
      return true;
    }
  });
}

const config = createLoggingProxy({ theme: 'dark', lang: 'en' }, 'Config');
config.theme;         // [GET] Config.theme
config.lang = 'fr';  // [SET] Config.lang = "fr"

// ============================
// Reactive State Proxy (simplified Vue 3 reactivity)
// ============================
function reactive(obj) {
  const listeners = new Set();

  return new Proxy(obj, {
    set(target, prop, value) {
      if (target[prop] !== value) {
        target[prop] = value;
        listeners.forEach(fn => fn());
      }
      return true;
    },
    get(target, prop) {
      if (prop === 'subscribe') {
        return (fn) => { listeners.add(fn); return () => listeners.delete(fn); };
      }
      return target[prop];
    }
  });
}

const state = reactive({ count: 0, name: 'App' });
const unsub = state.subscribe(() => console.log(`count=${state.count}`));
state.count = 1; // count=1
state.count = 2; // count=2
unsub();
```
