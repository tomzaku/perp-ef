The Mediator Pattern defines an object that encapsulates how a set of objects interact. Instead of objects communicating directly (creating a tangled web of dependencies), they communicate through a central mediator.

**Frontend use cases:**
- Chat room (users communicate through the room, not directly)
- Form coordination (fields validate based on other fields' values)
- UI component coordination (toolbar, sidebar, content interact through a mediator)
- Event bus / message broker

**Key difference from Observer:**
- Observer: subject notifies subscribers, but subscribers don't communicate through the subject
- Mediator: central hub that coordinates interaction BETWEEN multiple colleagues

**Benefit:** Reduces coupling — components only know about the mediator, not each other.

## Solution

```js
// ============================
// Chat Room Mediator
// ============================
class ChatRoom {
  constructor(name) {
    this.name = name;
    this.users = new Map();
    this.messageHistory = [];
  }

  join(user) {
    this.users.set(user.name, user);
    user.room = this;
    this.broadcast(`${user.name} joined`, 'System');
  }

  leave(user) {
    this.users.delete(user.name);
    user.room = null;
    this.broadcast(`${user.name} left`, 'System');
  }

  sendMessage(message, fromUser, toUserName) {
    this.messageHistory.push({
      from: fromUser.name, to: toUserName || 'all', message, timestamp: Date.now(),
    });

    if (toUserName) {
      const toUser = this.users.get(toUserName);
      if (toUser) toUser.receive(message, fromUser.name);
    } else {
      this.users.forEach((user, name) => {
        if (name !== fromUser.name) user.receive(message, fromUser.name);
      });
    }
  }

  broadcast(message, fromName) {
    this.users.forEach(user => user.receive(message, fromName));
  }
}

class ChatUser {
  constructor(name) {
    this.name = name;
    this.room = null;
    this.messages = [];
  }

  send(message, toUserName) {
    if (!this.room) throw new Error('Not in a room');
    this.room.sendMessage(message, this, toUserName);
  }

  receive(message, fromName) {
    const entry = `[${fromName}]: ${message}`;
    this.messages.push(entry);
    console.log(`${this.name} got — ${entry}`);
  }
}

const room = new ChatRoom('General');
const alice = new ChatUser('Alice');
const bob = new ChatUser('Bob');
const charlie = new ChatUser('Charlie');

room.join(alice);
room.join(bob);
room.join(charlie);

alice.send('Hello everyone!');
// Bob got — [Alice]: Hello everyone!
// Charlie got — [Alice]: Hello everyone!

bob.send('Hey Alice!', 'Alice');
// Alice got — [Bob]: Hey Alice!
// Charlie does NOT receive — it's a DM

// ============================
// Form Mediator — coordinate field interactions
// ============================
class FormMediator {
  constructor() {
    this.fields = new Map();
    this.rules = [];
  }

  registerField(name, field) {
    this.fields.set(name, field);
    field.mediator = this;
  }

  addRule(rule) { this.rules.push(rule); }

  fieldChanged(name, value) {
    this.rules.forEach(rule => rule(name, value, this.fields));
  }

  getValues() {
    const values = {};
    this.fields.forEach((field, name) => { values[name] = field.value; });
    return values;
  }
}

class FormField {
  constructor(name, value = '') {
    this.name = name;
    this.value = value;
    this.disabled = false;
    this.visible = true;
    this.mediator = null;
  }

  setValue(value) {
    this.value = value;
    if (this.mediator) this.mediator.fieldChanged(this.name, value);
  }
}

const form = new FormMediator();
const country = new FormField('country');
const state = new FormField('state');
const phone = new FormField('phone');

form.registerField('country', country);
form.registerField('state', state);
form.registerField('phone', phone);

// Rule: show/hide state based on country
form.addRule((name, value, fields) => {
  if (name === 'country') {
    fields.get('state').visible = (value === 'US' || value === 'CA');
    console.log(`State visible: ${fields.get('state').visible}`);
  }
});

// Rule: update phone format
form.addRule((name, value, fields) => {
  if (name === 'country') {
    const ph = fields.get('phone');
    ph.placeholder = value === 'US' ? '(555) 555-5555' : '+XX XXX XXX XXXX';
    console.log(`Phone format: ${ph.placeholder}`);
  }
});

country.setValue('US');
// State visible: true
// Phone format: (555) 555-5555

country.setValue('UK');
// State visible: false
// Phone format: +XX XXX XXX XXXX
```
