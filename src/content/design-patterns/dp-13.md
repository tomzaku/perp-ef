The Decorator Pattern attaches additional responsibilities to an object dynamically. It provides a flexible alternative to subclassing for extending functionality.

**Frontend use cases:**
- Higher-order functions (adding logging, timing, memoization)
- Middleware (Express, Redux)
- Adding behavior to components without modifying them (HOCs in React)
- Wrapping API calls with retry, caching, or authentication logic

**In JS, decorators are naturally expressed as:**
- Higher-order functions: `const enhanced = withLogging(originalFn)`
- Wrapper objects that delegate to the original
- TC39 decorator proposal for class methods

## Solution

```js
// ============================
// Function Decorators (higher-order functions)
// ============================
function withLogging(fn, label) {
  return function(...args) {
    console.log(`[${label}] Called with:`, args);
    const result = fn.apply(this, args);
    console.log(`[${label}] Returned:`, result);
    return result;
  };
}

function withTiming(fn, label) {
  return function(...args) {
    const start = performance.now();
    const result = fn.apply(this, args);
    console.log(`[${label}] took ${(performance.now() - start).toFixed(2)}ms`);
    return result;
  };
}

function withMemoization(fn) {
  const cache = new Map();
  return function(...args) {
    const key = JSON.stringify(args);
    if (cache.has(key)) return cache.get(key);
    const result = fn.apply(this, args);
    cache.set(key, result);
    return result;
  };
}

// Stack decorators
function add(a, b) { return a + b; }
const enhanced = withTiming(withLogging(withMemoization(add), 'add'), 'add');
console.log(enhanced(2, 3)); // Logged, timed, memoized

// ============================
// Object Decorator: wrapping an API client
// ============================
class BasicApiClient {
  async request(url, options = {}) {
    const res = await fetch(url, options);
    return res.json();
  }
}

function withAuth(client, getToken) {
  return {
    async request(url, options = {}) {
      return client.request(url, {
        ...options,
        headers: { ...options.headers, Authorization: `Bearer ${getToken()}` },
      });
    }
  };
}

function withRetry(client, maxRetries = 3) {
  return {
    async request(url, options = {}) {
      let lastError;
      for (let i = 0; i <= maxRetries; i++) {
        try {
          return await client.request(url, options);
        } catch (err) {
          lastError = err;
          console.log(`Retry ${i + 1}/${maxRetries}`);
          await new Promise(r => setTimeout(r, 1000 * (i + 1)));
        }
      }
      throw lastError;
    }
  };
}

function withCache(client, ttlMs = 60000) {
  const cache = new Map();
  return {
    async request(url, options = {}) {
      if (options.method && options.method !== 'GET') return client.request(url, options);
      const cached = cache.get(url);
      if (cached && Date.now() - cached.ts < ttlMs) return cached.data;
      const data = await client.request(url, options);
      cache.set(url, { data, ts: Date.now() });
      return data;
    }
  };
}

// Compose decorators — each adds a layer
let apiClient = new BasicApiClient();
apiClient = withCache(apiClient, 30000);
apiClient = withRetry(apiClient, 2);
apiClient = withAuth(apiClient, () => 'my-token');
// Final client has auth + retry + caching, core untouched
```
